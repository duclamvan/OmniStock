## Quick Integration for Replit

```javascript
// Add to your .env
TRACK17_API_KEY=your_api_key_here

// Simple tracking endpoint
app.post('/api/track-shipment', async (req, res) => {
  const { trackingNumber, carrier } = req.body;
  
  try {
    // Register for tracking
    const response = await track17.post('/register', {
      number: trackingNumber,
      carrier: carrier || null,
      auto_detect: !carrier
    });
    
    // Get tracking info
    const tracking = await track17.post('/gettrackinfo', {
      numbers: [trackingNumber]
    });
    
    res.json(tracking.data);
  } catch (error) {
    console.error('Tracking failed:', error);
    res.status(500).json({ error: 'Tracking failed' });
  }
});
```
